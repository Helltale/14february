cd (Desktop/example/) [заходим в папку нашего проекта]

git touch (name.txt) [создание пустого файла через коносль]
git status [определение состояния git'а]
git add (example.txt)[добавление файла]
git add -A [добавление всех файлов]
==========
git commit -m "(Комментарий к commit'у)"
==========
git remote add (название репозитория, origin) (ccылка URL, www.github.com/...) [подключение к удалённому репозиторию]
==========
git push (название репозитория, origin) (ветка изменения, master) [отправка измененений на сервер]
